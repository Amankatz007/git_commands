######## Basic command always used ###########
############################################## 
git add .                                     - To add any changes
git commit -m "<abc>"                         - To commit any changes
git push                                      - To push any changes to repository
git status                                    - To check and changes
git pull                                      - To pull any changes from repository
##############################################

# To clone a repository to a local folder
git clone <repository_link>

# To copy some other people public repository to our repository
--- Use the fork option in their repository

# To copy your local folder to git and make a repository from cmd
 1. Make a repository in GitHub (web interface)
 2. Go to your folder where you have files to upload and use the below commands
 3. git init     (to initialize the git in particular folder)
 4. git add .   (add all the files to staging git)
 5. git commit -m "Initial commit"  (make the commit to staging)
 6. git remote add origin <http url of the git repository created in point 1 >     (to make connection with git repository) 
 7. git push    (pushing the code to repository, might throw error if main is not set in upstream)
 8. (optional) git push --set-upstream origin main
 
# To pull the changes from git repository
git pull

# To get the updates on the files or repository 
git status    (very helpful to get the next suggestion steps and what is been changed)
git diff      (to track what changes have been made in the files)
git diff --cached (to check the changes after adding the files but not commited)

# To rename a file
git mv <file_name> <new_file_name>

# To go back in file before the commit
git restore .   (it will make the files as they were before you commit)
git restore --staged . (it will only help when you have added the files but not commited)

# To go back to previous version or check the commit history
git log   (to list all the commits, will provide the details of commit and checksum)
git log -p (will provide the descriptive list we have commited)
git log --oneline  (to get oneliner list of commit)

git show <checksum of specific version>         (will provide details of specific version)
git checkout <checksum of specific version>      (will take you to specific version but only for check)
git checkout main              (will bring you to the current working version)
